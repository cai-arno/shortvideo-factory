### 1.2 数据表 SQL

```sql
-- =============================================
-- 1. 租户表 (Tenant)
-- =============================================
CREATE TABLE tenants (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(100) NOT NULL,                    -- 租户/团队名称
    code VARCHAR(50) UNIQUE NOT NULL,               -- 租户代码 (用于URL等)
    plan VARCHAR(20) DEFAULT 'free',                -- free/pro/enterprise
    settings JSONB DEFAULT '{}',                   -- 租户配置
    status VARCHAR(20) DEFAULT 'active',           -- active/suspended
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- =============================================
-- 2. 用户表 (User) - 支持多端
-- =============================================
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    
    -- 账号基础
    phone VARCHAR(20) UNIQUE,                      -- 手机号 (主登录方式)
    email VARCHAR(255) UNIQUE,
    
    -- 微信相关
    union_id VARCHAR(100) UNIQUE,                  -- 微信 UnionID (跨应用)
    mini_program_openid VARCHAR(100),              -- 小程序 OpenID
    
    -- 账号状态
    status VARCHAR(20) DEFAULT 'active',            -- active/banned/pending
    last_login_at TIMESTAMP,
    
    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- =============================================
-- 3. 租户成员表 (TenantMember)
-- =============================================
CREATE TABLE tenant_members (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id UUID NOT NULL REFERENCES tenants(id) ON DELETE CASCADE,
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    role_id UUID NOT NULL REFERENCES roles(id),
    
    -- 成员配置
    nickname VARCHAR(50),                          -- 在团队内的昵称
    settings JSONB DEFAULT '{}',                   -- 个人设置
    
    -- 状态
    status VARCHAR(20) DEFAULT 'active',           -- active/pending/removed
    
    -- 时间戳
    joined_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(tenant_id, user_id)
);

-- =============================================
-- 4. 角色表 (Role)
-- =============================================
CREATE TABLE roles (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE,  -- NULL表示系统内置角色
    name VARCHAR(50) NOT NULL,                     -- 角色名
    code VARCHAR(50) NOT NULL,                     -- 角色代码
    description VARCHAR(200),
    is_system BOOLEAN DEFAULT FALSE,                -- 是否系统内置
    permissions JSONB DEFAULT '[]',                -- 权限列表 (冗余字段，加速查询)
    sort_order INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(tenant_id, code)
);

-- =============================================
-- 5. 权限表 (Permission)
-- =============================================
CREATE TABLE permissions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    code VARCHAR(100) UNIQUE NOT NULL,             -- 权限代码
    name VARCHAR(100) NOT NULL,                     -- 权限名称
    module VARCHAR(50) NOT NULL,                    -- 模块: scripts/videos/materials/templates/publishing/analytics/settings
    description VARCHAR(200),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- =============================================
-- 6. 角色-权限关联表 (RolePermission)
-- =============================================
CREATE TABLE role_permissions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    role_id UUID NOT NULL REFERENCES roles(id) ON DELETE CASCADE,
    permission_id UUID NOT NULL REFERENCES permissions(id) ON DELETE CASCADE,
    
    UNIQUE(role_id, permission_id)
);

-- =============================================
-- 7. 微信绑定表 (WechatBinding)
-- =============================================
CREATE TABLE wechat_bindings (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    union_id VARCHAR(100) NOT NULL,
    mini_program_openid VARCHAR(100),
    bind_type VARCHAR(20) DEFAULT 'auto',           -- auto(静默绑定)/manual(手动绑定)
    bound_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(union_id)
);

-- =============================================
-- 8. 登录设备表 (UserDevice)
-- =============================================
CREATE TABLE user_devices (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    device_id VARCHAR(100) NOT NULL,                -- 设备标识
    device_type VARCHAR(20) NOT NULL,               -- mobile/pc/web/wx_mini
    device_name VARCHAR(100),                       -- 设备名称
    last_login_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    UNIQUE(user_id, device_id)
);

-- =============================================
-- 9. 短信验证码表 (SMSCode) - 临时表
-- =============================================
CREATE TABLE sms_codes (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    phone VARCHAR(20) NOT NULL,
    code VARCHAR(10) NOT NULL,
    usage VARCHAR(20) DEFAULT 'login',             -- login/register/bind/reset
    expires_at TIMESTAMP NOT NULL,
    used_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    INDEX idx_phone_usage (phone, usage),
    INDEX idx_expires (expires_at)
);
```

### 1.3 业务表添加 tenant_id

所有业务表需添加 `tenant_id` 字段：

```sql
-- 示例：scripts 表
ALTER TABLE scripts ADD COLUMN tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE;
ALTER TABLE scripts ADD COLUMN creator_id UUID REFERENCES users(id);
CREATE INDEX idx_scripts_tenant ON scripts(tenant_id);

-- videos 表
ALTER TABLE videos ADD COLUMN tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE;
ALTER TABLE videos ADD COLUMN creator_id UUID REFERENCES users(id);
CREATE INDEX idx_videos_tenant ON videos(tenant_id);

-- materials 表
ALTER TABLE materials ADD COLUMN tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE;
ALTER TABLE materials ADD COLUMN uploader_id UUID REFERENCES users(id);
CREATE INDEX idx_materials_tenant ON materials(tenant_id);

-- templates 表
ALTER TABLE templates ADD COLUMN tenant_id UUID REFERENCES tenants(id) ON DELETE CASCADE;
ALTER TABLE templates ADD COLUMN creator_id UUID REFERENCES users(id);
CREATE INDEX idx_templates_tenant ON templates(tenant_id);
```

```python
# app/models/user.py
from datetime import datetime
from typing import Optional, List
from sqlmodel import SQLModel, Field, Relationship
import uuid


class User(SQLModel, table=True):
    __tablename__ = "users"
    
    id: Optional[uuid.UUID] = Field(default_factory=uuid.uuid4, primary_key=True)
    
    # 账号基础
    phone: Optional[str] = Field(default=None, unique=True, max_length=20)
    email: Optional[str] = Field(default=None, unique=True, max_length=255)
    
    # 微信
    union_id: Optional[str] = Field(default=None, unique=True, max_length=100)
    mini_program_openid: Optional[str] = Field(default=None, max_length=100)
    
    # 密码 (如需密码登录)
    password_hash: Optional[str] = Field(default=None, max_length=255)
    
    # 状态
    status: str = Field(default="active", max_length=20)
    last_login_at: Optional[datetime] = Field(default=None)
    
    # 时间戳
    created_at: datetime = Field(default_factory=datetime.utcnow)
    updated_at: datetime = Field(default_factory=datetime.utcnow)
    
    # 关系
    tenant_memberships: List["TenantMember"] = Relationship(back_populates="user")


class Tenant(SQLModel, table=True):
    __tablename__ = "tenants"
    
    id: Optional[uuid.UUID] = Field(default_factory=uuid.uuid4, primary_key=True)
    name: str = Field(max_length=100)
    code: str = Field(max_length=50, unique=True)
    plan: str = Field(default="free", max_length=20)
    settings: dict = Field(default={})
    status: str = Field(default="active", max_length=20)
    created_at: datetime = Field(default_factory=datetime.utcnow)
    updated_at: datetime = Field(default_factory=datetime.utcnow)
    
    # 关系
    members: List["TenantMember"] = Relationship(back_populates="tenant")


class TenantMember(SQLModel, table=True):
    __tablename__ = "tenant_members"
    
    id: Optional[uuid.UUID] = Field(default_factory=uuid.uuid4, primary_key=True)
    tenant_id: uuid.UUID = Field(foreign_key="tenants.id")
    user_id: uuid.UUID = Field(foreign_key="users.id")
    role_id: uuid.UUID = Field(foreign_key="roles.id")
    
    nickname: Optional[str] = Field(default=None, max_length=50)
    settings: dict = Field(default={})
    status: str = Field(default="active", max_length=20)
    joined_at: datetime = Field(default_factory=datetime.utcnow)
    created_at: datetime = Field(default_factory=datetime.utcnow)
    updated_at: datetime = Field(default_factory=datetime.utcnow)
    
    # 关系
    tenant: Optional[Tenant] = Relationship(back_populates="members")
    user: Optional[User] = Relationship(back_populates="tenant_memberships")
    role: Optional["Role"] = Relationship()


class Role(SQLModel, table=True):
    __tablename__ = "roles"
    
    id: Optional[uuid.UUID] = Field(default_factory=uuid.uuid4, primary_key=True)
    tenant_id: Optional[uuid.UUID] = Field(default=None, foreign_key="tenants.id")
    name: str = Field(max_length=50)
    code: str = Field(max_length=50)
    description: Optional[str] = Field(default=None, max_length=200)
    is_system: bool = Field(default=False)
    permissions: List[str] = Field(default=[])
    sort_order: int = Field(default=0)
    created_at: datetime = Field(default_factory=datetime.utcnow)
    updated_at: datetime = Field(default_factory=datetime.utcnow)


class Permission(SQLModel, table=True):
    __tablename__ = "permissions"
    
    id: Optional[uuid.UUID] = Field(default_factory=uuid.uuid4, primary_key=True)
    code: str = Field(max_length=100, unique=True)
    name: str = Field(max_length=100)
    module: str = Field(max_length=50)
    description: Optional[str] = Field(default=None, max_length=200)
    created_at: datetime = Field(default_factory=datetime.utcnow)
```

#### 3.2.2 租户上下文
```python
# app/core/tenant.py
from dataclasses import dataclass
from datetime import datetime
from typing import Optional, List
from uuid import UUID
import uuid


@dataclass
class TenantContext:
    """租户上下文 - 存储当前请求的租户信息"""
    tenant_id: UUID
    user_id: UUID
    role_id: UUID
    permissions: List[str]
    
    @property
    def is_admin(self) -> bool:
        return "*" in self.permissions or "admin" in self.permissions


@dataclass  
class CurrentUser:
    """当前登录用户"""
    id: UUID
    phone: str
    union_id: Optional[str] = None
    
    @property
    def display_phone(self) -> str:
        """脱敏手机号"""
        if self.phone:
            return self.phone[:3] + "****" + self.phone[-4:]
        return ""


# 线程本地存储
import contextvars
tenant_context: contextvars.ContextVar[Optional[TenantContext]] = contextvars.ContextVar(
    "tenant_context", default=None
)

current_user: contextvars.ContextVar[Optional[CurrentUser]] = contextvars.ContextVar(
    "current_user", default=None
)


def get_tenant_context() -> Optional[TenantContext]:
    return tenant_context.get()


def require_tenant() -> TenantContext:
    """获取租户上下文，如果不存在则抛出异常"""
    ctx = tenant_context.get()
    if not ctx:
        raise ValueError("No tenant context")
    return ctx
```

```python
from datetime import datetime, timedelta
from typing import Optional, List
from uuid import UUID
import jwt
from passlib.context import CryptContext
from app.core.config import settings
from app.core.tenant import TenantContext, CurrentUser


pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")


def create_access_token(
    user_id: UUID,
    tenant_id: UUID,
    role_id: UUID,
    permissions: List[str],
    expires_delta: Optional[timedelta] = None
) -> str:
    """创建 JWT Access Token"""
    if expires_delta is None:
        expires_delta = timedelta(minutes=settings.ACCESS_TOKEN_EXPIRE_MINUTES)
    
    expire = datetime.utcnow() + expires_delta
    
    payload = {
        "sub": str(user_id),
        "tenant_id": str(tenant_id),
        "role_id": str(role_id),
        "permissions": permissions,
        "exp": expire,
        "iat": datetime.utcnow(),
    }
    
    return jwt.encode(payload, settings.SECRET_KEY, algorithm=settings.ALGORITHM)


def decode_token(token: str) -> Optional[dict]:
    """解码 JWT Token"""
    try:
        payload = jwt.decode(
            token, 
            settings.SECRET_KEY, 
            algorithms=[settings.ALGORITHM]
        )
        return payload
    except jwt.ExpiredSignatureError:
        return None
    except jwt.InvalidTokenError:
        return None


def hash_password(password: str) -> str:
    """哈希密码"""
    return pwd_context.hash(password)


def verify_password(plain_password: str, hashed_password: str) -> bool:
    """验证密码"""
    return pwd_context.verify(plain_password, hashed_password)
```

```python
# app/core/deps.py
from fastapi import Depends, HTTPException, Header
from sqlalchemy.ext.asyncio import AsyncSession
from typing import Optional, List
from uuid import UUID
import uuid

from app.core.database import async_session
from app.core.security import decode_token
from app.core.tenant import TenantContext, CurrentUser, tenant_context, current_user
from app.models.user import User, Tenant, TenantMember, Role


async def get_db() -> AsyncSession:
    async with async_session() as session:
        yield session


async def get_current_user(
    authorization: Optional[str] = Header(None)
) -> User:
    """获取当前用户（从Token）"""
    if not authorization:
        raise HTTPException(status_code=401, detail="未登录")
    
    if authorization.startswith("Bearer "):
        token = authorization[7:]
    else:
        token = authorization
    
    payload = decode_token(token)
    if not payload:
        raise HTTPException(status_code=401, detail="Token无效或已过期")
    
    user_id = payload.get("sub")
    if not user_id:
        raise HTTPException(status_code=401, detail="Token无效")
    
    async with async_session() as session:
        user = await session.get(User, uuid.UUID(user_id))
        if not user:
            raise HTTPException(status_code=401, detail="用户不存在")
        
        return user


async def get_tenant_context(
    user: User = Depends(get_current_user)
) -> TenantContext:
    """获取租户上下文 - 从请求头或默认租户"""
    # TODO: 支持从请求头 X-Tenant-ID 读取，或者从用户默认租户
    async with async_session() as session:
        # 获取用户的第一个租户成员关系
        member = await session.query(TenantMember).filter(
            TenantMember.user_id == user.id,
            TenantMember.status == "active"
        ).first()
        
        if not member:
            raise HTTPException(status_code=403, detail="用户未加入任何团队")
        
        # 获取角色权限
        role = await session.get(Role, member.role_id)
        permissions = role.permissions if role else []
        
        ctx = TenantContext(
            tenant_id=member.tenant_id,
            user_id=user.id,
            role_id=member.role_id,
            permissions=permissions
        )
        
        # 设置上下文
        tenant_context.set(ctx)
        current_user.set(CurrentUser(id=user.id, phone=user.phone))
        
        return ctx


def require_permission(*required_permissions: str):
    """权限检查装饰器"""
    def dependency(ctx: TenantContext = Depends(get_tenant_context)):
        # 管理员拥有所有权限
        if "*" in ctx.permissions:
            return ctx
        
        # 检查权限
        for perm in required_permissions:
            # 支持通配符: scripts:* 表示 scripts 模块所有权限
            if perm.endswith(":*"):
                module = perm[:-2]
                has_perm = any(
                    p == perm or (p.startswith(module + ":") and p != module + ":read")
                    for p in ctx.permissions
                )
            else:
                has_perm = perm in ctx.permissions
            
            if not has_perm:
                raise HTTPException(status_code=403, detail=f"缺少权限: {perm}")
        
        return ctx
    
    return dependency
```

```python
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession

    code: str
    device_id: Optional[str] = None
    device_type: str = "web"


class LoginResponse(BaseModel):
    access_token: str
    token_type: str = "Bearer"
    expires_in: int
    user: dict
    tenant: Optional[dict] = None


@router.post("/sms/send")
async def send_sms(
    request: SendSMSRequest,
    db: AsyncSession = Depends(get_db)
):
    """发送验证码"""
    sms_service = SMSService()
    await sms_service.send_code(request.phone, request.usage)
    return {"success": True, "message": "验证码已发送"}


@router.post("/sms/login", response_model=LoginResponse)
async def sms_login(
    request: SMSLoginRequest,
    db: AsyncSession = Depends(get_db)
):
    """手机号验证码登录"""
    # 1. 验证验证码
    sms_service = SMSService()
    if not await sms_service.verify_code(request.phone, request.code, request.usage):
        raise HTTPException(status_code=400, detail="验证码错误或已过期")
    
    # 2. 查找或创建用户
    user = await db.query(User).filter(User.phone == request.phone).first()
    if not user:
        user = User(phone=request.phone)
        db.add(user)
        await db.flush()
        
        # 创建用户默认租户
        tenant = Tenant(name=f"{request.phone}的团队", code=f"user_{user.id}")
        db.add(tenant)
        await db.flush()
        
        # 默认角色
        viewer_role = await db.query(Role).filter(Role.code == "viewer").first()
        
        # 添加为租户管理员
        member = TenantMember(
            tenant_id=tenant.id,
            user_id=user.id,
            role_id=viewer_role.id,
            nickname="管理员"
        )
        db.add(member)
    
    # 3. 更新最后登录时间
    from datetime import datetime
    user.last_login_at = datetime.utcnow()
    await db.commit()
    
    # 4. 获取用户当前租户
    member = await db.query(TenantMember).filter(
        TenantMember.user_id == user.id,
        TenantMember.status == "active"
    ).first()
    
    if not member:
        raise HTTPException(status_code=403, detail="用户未加入任何团队")
    
    # 5. 获取权限
    role = await db.get(Role, member.role_id)
    permissions = role.permissions if role else []
    
    # 6. 创建 Token
    token = create_access_token(
        user_id=user.id,
        tenant_id=member.tenant_id,
        role_id=member.role_id,
        permissions=permissions
    )
    
    return LoginResponse(
        access_token=token,
        expires_in=7 * 24 * 3600,
        user={
            "id": str(user.id),
            "phone": user.phone[:3] + "****" + user.phone[-4:] if user.phone else None,
        },
        tenant={
            "id": str(member.tenant_id),
            "role": role.code if role else None
        }
    )


@router.get("/tenants")
async def list_user_tenants(
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db)
):
    """获取用户所有租户"""
    members = await db.query(TenantMember).filter(
        TenantMember.user_id == user.id,
        TenantMember.status == "active"
    ).all()
    
    items = []
    for m in members:
        tenant = await db.get(Tenant, m.tenant_id)
        role = await db.get(Role, m.role_id)
        if tenant:
            items.append({
                "id": str(tenant.id),
                "name": tenant.name,
                "role": role.code if role else None,
                "current": False  # TODO: 判断当前租户
            })
    
    return {"tenants": items}


@router.post("/tenants/{tenant_id}/switch")
async def switch_tenant(
    tenant_id: UUID,
    user: User = Depends(get_current_user),
    db: AsyncSession = Depends(get_db)
):
    """切换租户"""
    # 验证用户是否属于该租户
    member = await db.query(TenantMember).filter(
        TenantMember.user_id == user.id,
        TenantMember.tenant_id == tenant_id,
        TenantMember.status == "active"
    ).first()
    
    if not member:
        raise HTTPException(status_code=403, detail="无权访问该团队")
    
    # 获取角色和权限
    role = await db.get(Role, member.role_id)
    permissions = role.permissions if role else []
    
    # 创建新 Token
    token = create_access_token(
        user_id=user.id,
        tenant_id=member.tenant_id,
        role_id=member.role_id,
        permissions=permissions
    )
    
    return {
        "access_token": token,
        "tenant": {
            "id": str(tenant_id),
            "role": role.code if role else None
        }
    }
```

## 四、数据隔离策略总结

### 4.1 租户隔离层级

| 层级 | 策略 | 实现方式 |
|------|------|----------|
| **数据库层** | 独立 Schema 或独立数据库 | PostgreSQL Schema per tenant |
| **表级** | tenant_id 外键约束 | 所有业务表添加 tenant_id |
| **行级** | Row-Level Security (RLS) | PostgreSQL RLS 策略 |
| **应用层** | 中间件 + 查询过滤 | API 层统一校验 |

### 4.2 推荐方案

**轻量级（当前项目）**：应用层隔离
- 所有表添加 `tenant_id`
- API 层通过中间件自动注入和校验
- 查询时统一添加 `tenant_id` 条件

**进阶方案**：PostgreSQL RLS
```sql
-- 启用行级安全
ALTER TABLE scripts ENABLE ROW LEVEL SECURITY;

-- 创建策略
CREATE POLICY tenant_isolation ON scripts
    USING (tenant_id = current_setting('app.current_tenant')::uuid);
```

### 4.3 权限检查流程

```
请求 → JWT Token → 解析 user_id/tenant_id/permissions
       ↓
   设置 TenantContext
       ↓
   require_permission("scripts:create")
       ↓
   检查 ctx.permissions 是否包含
       ↓
   业务逻辑 → 查询自动附加 tenant_id 过滤
       ↓
   返回结果
```
